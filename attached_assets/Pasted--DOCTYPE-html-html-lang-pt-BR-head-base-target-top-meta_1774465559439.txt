    <script>
        function toggleSidebar() {
            const sb = document.getElementById('sidebar');
            sb.classList.toggle('collapsed');
        }

        function abrirDashboardProfs() {
            document.getElementById('dashboard-profs').style.display = 'block';
            document.getElementById('perfil-detalhado').style.display = 'none';
        }

        function verPerfilCompleto(prof) {
            document.getElementById('dashboard-profs').style.display = 'none';
            document.getElementById('perfil-detalhado').style.display = 'block';

            // Preenche os dados
            document.getElementById('p-nome').innerText = prof.nome;
            document.getElementById('p-vínculo').innerText = prof.vinculo || "Professor";
            document.getElementById('p-cpf').innerText = prof.cpf;
            document.getElementById('p-matricula').innerText = prof.matricula;
            document.getElementById('p-turmaM').innerText = prof.turmaManha;
            document.getElementById('p-turmaT').innerText = prof.turmaTarde;
            document.getElementById('p-turno').innerText = prof.turno;
            document.getElementById('p-fone').innerText = prof.telefone;
            document.getElementById('p-email').innerText = prof.email;
            document.getElementById('p-censo').innerText = prof.censo;
            document.getElementById('p-nascimento').innerText = prof.nascimento;
            document.getElementById('p-foto').src = prof.foto || "https://www.w3schools.com/howto/img_avatar.png";
        }

        window.onload = function() {
            const recentGrid = document.getElementById('recent-list');

            professoresExemplo.forEach((prof, index) => {
                // Preenche Recentes (4 primeiros)
