    <script>
        let nomesAlunos = [];
        window.onload = function() {
            google.script.run.withSuccessHandler(function(nomes) {
                nomesAlunos = nomes;
                const dl = document.getElementById('listaAlunos');
                nomes.forEach(n => { const opt = document.createElement('option'); opt.value = n; dl.appendChild(opt); });
            }).listarNomesAlunos();
        };

        function verificarInput(val) { if(nomesAlunos.includes(val)) pesquisar(); }

        function pesquisar() {
            const busca = document.getElementById('inputBusca').value;
            if(!busca) return;
            document.getElementById('msgStatus').innerText = "Buscando registro...";
            google.script.run.withSuccessHandler(function(aluno) {
                if(aluno) {
                    document.getElementById('msgStatus').style.display = 'none';
                    document.getElementById('areaPerfil').style.display = 'grid';
                    document.getElementById('val-Matricula').innerText = aluno["Matrícula"] || "---";
                    document.getElementById('val-Turma').innerText = aluno["Turma Atual"] || "---";
                    document.getElementById('val-Turno').innerText = aluno["Turno"] || "---";
                    document.getElementById('val-Situacao').innerText = aluno["Situação no Curso"] || "---";
                    document.getElementById('val-Nome').innerText = aluno["Nome do aluno"] || "---";
                    document.getElementById('val-Nascimento').innerText = aluno["Data de Nascimento"] || "---";
                    document.getElementById('val-CPF').innerText = aluno["CPF"] || "---";
                    document.getElementById('val-RG').innerText = aluno["RG"] || "---";
                    document.getElementById('val-Mae').innerText = aluno["Nome da mãe"] || "---";
                    document.getElementById('val-Pai').innerText = aluno["Nome do pai"] || "---";
                    document.getElementById('val-Telefone').innerText = aluno["Telefone"] || "---";
                    document.getElementById('val-Endereco').innerText = aluno["Endereço"] || "---";
                } else { 
                    document.getElementById('areaPerfil').style.display = 'none';
                    document.getElementById('msgStatus').style.display = 'block';
                    document.getElementById('msgStatus').innerText = "❌ Aluno não encontrado.";
                }
            }).obterDadosAluno(busca);
        }
        function voltar() { google.script.run.withSuccessHandler(url => window.top.location.href = url + "?page=Dashboard").pegarLinkDoApp(); }
    </script>
