    <script>
        window.onload = function() {
            google.script.run.withSuccessHandler(turmas => {
                const s = document.getElementById('selTurma');
                s.innerHTML = '<option value="">Selecione a Turma...</option>';
                turmas.forEach(t => { 
                    let o = document.createElement('option'); o.value = t; o.text = "TURMA: " + t; s.add(o); 
                });
            }).listarTurmasUnicas();
        };

        function prepararGeracao() {
            const turma = document.getElementById('selTurma').value;
            if(!turma) return alert("Por favor, selecione uma turma primeiro!");

            // Captura opções do Cabeçalho
            const showProfessor = document.getElementById('headProfessor').checked;
            const showTurma = document.getElementById('headTurma').checked;

            // Captura quais colunas da tabela foram marcadas
            const checkboxes = document.querySelectorAll('#areaChecks input:checked');
            let colunasEscolhidas = [{label: "Nº", ref: "n"}, {label: "NOME DO ALUNO", ref: "Nome do aluno"}];
            
            checkboxes.forEach(ck => {
                if(ck.id === 'assinatura') colunasEscolhidas.push({label: "ASSINATURA", ref: "vazio"});
                else colunasEscolhidas.push({label: ck.parentElement.innerText.trim().toUpperCase(), ref: ck.getAttribute('data-ref')});
            });

            // Chama o servidor para buscar os alunos E o nome do professor
            google.script.run.withSuccessHandler(function(alunos) {
                if(!alunos || alunos.length === 0) return alert("Nenhum aluno encontrado para (" + turma + ").");
                
                if(showProfessor) {
                    // Se pediu professor, faz uma segunda chamada para buscar o nome dele
                    google.script.run.withSuccessHandler(function(nomeProfessor) {
                        abrirJanelaImpressao(turma, colunasEscolhidas, alunos, {showProfessor, showTurma, nomeProfessor});
                    }).obterProfessorDaTurma(turma);
                } else {
                    // Se não pediu professor, gera direto
                    abrirJanelaImpressao(turma, colunasEscolhidas, alunos, {showProfessor, showTurma, nomeProfessor: ""});
                }
                
            }).buscarAlunosPorTurma(turma);
        }
