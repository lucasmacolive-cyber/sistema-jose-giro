    <script>
        // Formata a data na Topbar automaticamente
        document.getElementById('data-top').innerText = new Date().toLocaleDateString('pt-BR', {
            weekday: 'long', day: 'numeric', month: 'long'
        });

        function realizarLogin() {
            const user = document.getElementById('usuario').value.trim();
            const pass = document.getElementById('senha').value.trim();
            const msg = document.getElementById('mensagem');
            const btn = document.getElementById('btnEntrar');

            if (!user || !pass) {
                msg.className = "erro";
                msg.innerText = "⚠️ Por favor, preencha todos os campos.";
                return;
            }

            // Inicia estado visual de carregamento
            btn.innerText = "VERIFICANDO...";
            btn.disabled = true;
            msg.innerText = "";

            // COMUNICAÇÃO COM O SERVIDOR (Login.gs -> processarLogin)
            google.script.run.withSuccessHandler(function(res) {
                if (res.sucesso) {
                    msg.className = "sucesso";
                    msg.innerText = "✅ Acesso permitido! Entrando...";
                    
                    // Guarda os dados para usar no Dashboard
                    localStorage.setItem('nomeUsuario', res.nome);
                    localStorage.setItem('generoUsuario', res.genero);
                    
                    // Busca a URL do App e redireciona
                    google.script.run.withSuccessHandler(function(url) {
                        window.top.location.href = url + "?page=Dashboard";
                    }).pegarLinkDoApp();
                } else {
                    msg.className = "erro";
                    msg.innerText = "❌ " + (res.mensagem || "Login ou senha incorretos.");
                    btn.innerText = "ENTRAR NO SISTEMA";
                    btn.disabled = false;
                }
            }).withFailureHandler(function(err) {
                msg.className = "erro";
                msg.innerText = "❌ Erro crítico no servidor. Tente novamente.";
                btn.innerText = "ENTRAR NO SISTEMA";
                btn.disabled = false;
            }).processarLogin(user, pass); // Chama a função correta no Login.gs
        }

        // Atalho teclado: Tecla Enter aciona o botão
        document.addEventListener('keypress', function(e) {
            if (e.key === 'Enter') realizarLogin();
        });
    </script>
