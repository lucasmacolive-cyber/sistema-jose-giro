    <script>
        function toggleSidebar() {
            const side = document.getElementById('sidebar');
            const main = document.getElementById('main');
            const seta = document.getElementById('btnSeta');
            side.classList.toggle('oculta');
            main.classList.toggle('expandido');
            seta.innerText = side.classList.contains('oculta') ? "▶" : "◀";
        }

        function atualizarInfo() {
            const dias = ['Domingo','Segunda','Terça','Quarta','Quinta','Sexta','Sábado'];
            const meses = ['Janeiro','Fevereiro','Março','Abril','Maio','Junho','Julho','Agosto','Setembro','Outubro','Novembro','Dezembro'];
            const agora = new Date();
            document.getElementById('data-hoje').innerText = `${dias[agora.getDay()]}, ${agora.getDate()} de ${meses[agora.getMonth()]}`;
            
            const hora = agora.getHours();
            let s = hora < 5 ? "Boa madrugada" : hora < 12 ? "Bom dia" : hora < 18 ? "Boa tarde" : "Boa noite";
            document.getElementById('txtSaudacao').innerText = s + ",";

            const nome = localStorage.getItem('nomeUsuario');
            const gen = localStorage.getItem('generoUsuario');
            if (nome && nome !== "null") {
                const prefixo = (gen === 'F' || gen === 'FEMININO') ? 'Tia' : 'Tio';
                document.getElementById('txtNomeUsuario').innerText = `${prefixo} ${nome.split(' ')[0]}`;
            }
        }

        async function obterClima() {
            try {
                const res = await fetch('https://ipapi.co/json/');
                const local = await res.json();
                document.getElementById('clima-local').innerText = `☀️ 26°C em ${local.city}`;
            } catch (e) { document.getElementById('clima-local').innerText = "☀️ 27°C"; }
        }

        function navegar(pg) { 
            google.script.run.withSuccessHandler(url => window.top.location.href = url + "?page=" + pg).pegarLinkDoApp(); 
        }
        
        function sair() { 
            localStorage.clear(); 
            google.script.run.withSuccessHandler(url => window.top.location.href = url).pegarLinkDoApp(); 
        }

        window.onload = function() { 
            atualizarInfo(); 
            obterClima(); 
        };
    </script>
