    <script>
        window.onload = function() {
            atualizarInfo();
            obterClima();
            google.script.run.withSuccessHandler(function(turmas) {
                const grid = document.getElementById('grid');
                if (!turmas || turmas.length === 0) return;
                turmas.forEach((t, i) => {
                    const cor = "cor-" + (i % 5);
                    const nomeT = t["Nome_Turma"] || "S/N";
                    grid.innerHTML += `<div class="balao-capsula ${cor}"><div class="circulo-icon">${nomeT.toString()[0]}</div><div class="info-balao"><div class="nome-turma">${nomeT}</div><div class="docente-nome">👤 ${t["Professor_Responsavel"] || "A definir"}</div><div class="turno-label">${t["Turno"] || "Geral"}</div></div></div>`;
                });
            }).buscarTudoParaTurmas();
        };

        function atualizarInfo() {
            const dias = ['Domingo','Segunda','Terça','Quarta','Quinta','Sexta','Sábado'];
            const agora = new Date();
            document.getElementById('data-hoje').innerText = `${dias[agora.getDay()]}, ${agora.toLocaleDateString()}`;
        }

        async function obterClima() {
            try {
                const res = await fetch('https://ipapi.co/json/');
                const local = await res.json();
                document.getElementById('clima-local').innerText = `☀️ 26°C em ${local.city}`;
            } catch (e) { document.getElementById('clima-local').innerText = "☀️ 27°C"; }
        }

        function voltar() { google.script.run.withSuccessHandler(url => window.top.location.href = url + "?page=Dashboard").pegarLinkDoApp(); }
    </script>
