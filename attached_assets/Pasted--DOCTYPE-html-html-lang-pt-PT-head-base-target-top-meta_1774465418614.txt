    <script>
        // Função para mostrar/esconder o menu no telemóvel
        function toggleMenu() {
            document.getElementById('sidebar').classList.toggle('active');
        }

        // Função para sair do sistema
        function fazerLogout() {
            localStorage.clear();
            // Pede ao Google o link do sistema para voltar ao Login
            google.script.run.withSuccessHandler(function(url) {
                window.top.location.href = url + "?page=Index";
            }).pegarLinkDoApp();
        }

        // Quando a página carregar, puxa o nome da pessoa da memória
        window.onload = function() {
            const nome = localStorage.getItem('usuarioLogado');
            const perfil = localStorage.getItem('perfilUsuario');
            const genero = localStorage.getItem('generoUsuario');

            const saudacao = document.getElementById('mensagem-boas-vindas');

            if (!nome) {
                // Se não houver ninguém logado, expulsa de volta para o login!
                fazerLogout();
            } else {
                // Monta a mensagem personalizada (Ex: Bem-vinda, Diretora Maria)
                let pronome = (genero === 'F' || genero === 'f') ? 'Bem-vinda' : 'Bem-vindo';
                saudacao.innerText = `${pronome}, ${nome} (${perfil})`;
            }
        }
    </script>
