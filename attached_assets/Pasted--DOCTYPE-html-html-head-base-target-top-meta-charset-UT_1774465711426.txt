    <script>
        window.onload = function() {
            google.script.run.withSuccessHandler(function(alunos) {
                const grid = document.getElementById('grid');
                if (!alunos || alunos.length === 0) return;

                alunos.forEach((a, i) => {
                    const cor = "cor-" + (i % 5);
                    const nomeA = a["Nome Completo"] || a["Nome_Aluno"] || "Estudante";
                    const turmaA = a["Turma"] || "S/T";

                    grid.innerHTML += `
                        <div class="balao-capsula ${cor}">
                            <div class="circulo-icon">${nomeA.toString()[0]}</div>
                            <div class="info-balao">
                                <div class="nome-aluno">${nomeA}</div>
                                <div class="tag-turma">${turmaA}</div>
                            </div>
                        </div>
                    `;
                });
            }).buscarAlunos();
        };

        function voltar() {
            google.script.run.withSuccessHandler(url => window.top.location.href = url + "?page=Dashboard").pegarLinkDoApp();
        }
    </script>
